Script SQL — Criar Tabelas no Supabase

Cole este script no SQL Editor do Supabase e clique em Run. Execute apenas uma vez. Se as tabelas já existirem, ignore. 

-- ============================================
-- SGH · Script de criação das tabelas
-- Cole no SQL Editor do Supabase e clique Run
-- ============================================

-- Tabela de sensores
CREATE TABLE IF NOT EXISTS sensores (
  id          SERIAL PRIMARY KEY,
  nome_sensor TEXT        NOT NULL,
  tipo_sensor TEXT        NOT NULL,
  localizacao TEXT        DEFAULT '',
  ativo       BOOLEAN     DEFAULT TRUE,
  criado_em   TIMESTAMPTZ DEFAULT NOW()
);

-- Tabela de medições (todas as leituras registradas)
CREATE TABLE IF NOT EXISTS medicoes (
  id          SERIAL PRIMARY KEY,
  id_sensor   INT         REFERENCES sensores(id) ON DELETE SET NULL,
  nome_sensor TEXT,
  tipo        TEXT        NOT NULL,
  valor       NUMERIC(10,2) NOT NULL,
  unidade     TEXT        DEFAULT '',
  status      TEXT        DEFAULT 'ok',   -- 'ok', 'atencao', 'critico'
  observacao  TEXT        DEFAULT '',
  data_hora   TEXT,
  criado_em   TIMESTAMPTZ DEFAULT NOW()
);

-- Índices para acelerar os filtros do histórico
CREATE INDEX IF NOT EXISTS idx_medicoes_tipo      ON medicoes(tipo);
CREATE INDEX IF NOT EXISTS idx_medicoes_sensor    ON medicoes(id_sensor);
CREATE INDEX IF NOT EXISTS idx_medicoes_status    ON medicoes(status);
CREATE INDEX IF NOT EXISTS idx_medicoes_criado_em ON medicoes(criado_em DESC);

-- Tabela de alertas (registrada automaticamente ao detectar fora do limite)
CREATE TABLE IF NOT EXISTS alertas (
  id          SERIAL PRIMARY KEY,
  id_sensor   INT         REFERENCES sensores(id) ON DELETE SET NULL,
  nome_sensor TEXT,
  tipo        TEXT        NOT NULL,
  valor       NUMERIC(10,2),
  nivel       TEXT        NOT NULL,        -- 'atencao' ou 'critico'
  mensagem    TEXT,
  resolvido   BOOLEAN     DEFAULT FALSE,
  criado_em   TIMESTAMPTZ DEFAULT NOW()
);

-- Tabela de configurações (limites e preferências)
CREATE TABLE IF NOT EXISTS configuracoes (
  chave     TEXT PRIMARY KEY,
  valor     JSONB,
  criado_em TIMESTAMPTZ DEFAULT NOW()
);
